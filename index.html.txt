  <script>
    const canvas = document.getElementById('confettiCanvas');
    const ctx = canvas.getContext('2d');

    let W, H, confetti = [], animationId;

    function resizeCanvas(){
      W = canvas.width = canvas.offsetWidth;
      H = canvas.height = canvas.offsetHeight;
    }
    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();

    function r(min,max){return Math.random()*(max-min)+min}

    function spawnConfetti(){
      for(let i=0;i<120;i++){
        confetti.push({
          x:r(0,W), y:r(-H,0),
          r:r(6,12), vx:r(-1.4,1.4),
          vy:r(2,6), tilt:r(-10,10),
          tiltSpeed:r(0.05,0.12),
          color:['#ff2d7a','#fff176','#8bc34a','#00bcd4','#ff8bcf'][Math.floor(r(0,5))]
        })
      }
      if(!animationId) animate();
    }

    function draw(){
      ctx.clearRect(0,0,W,H);
      for(let p of confetti){
        p.x+=p.vx; p.y+=p.vy; p.tilt+=p.tiltSpeed;
        ctx.save(); ctx.fillStyle=p.color;
        ctx.translate(p.x,p.y); ctx.rotate(p.tilt*Math.PI/180);
        ctx.fillRect(-p.r/2,-p.r/2,p.r,p.r*0.6);
        ctx.restore();
      }
      confetti = confetti.filter(p=>p.y<H+30);
      if(confetti.length===0){ cancelAnimationFrame(animationId); animationId=null }
    }

    function animate(){ animationId=requestAnimationFrame(animate); draw(); }

    document.getElementById('confettiBtn').onclick = ()=>{ spawnConfetti(); createSparkles(20); }

    const sparklesEl = document.getElementById('sparkles');
    function createSparkles(n){
      for(let i=0;i<n;i++){
        const s=document.createElement('div'); s.className='sparkle';
        s.style.left=(10+Math.random()*80)+"%";
        s.style.top=(10+Math.random()*80)+"%";
        sparklesEl.appendChild(s);
        const life=800+Math.random()*900;
        s.animate([{opacity:1,transform:'scale(1)'},{opacity:0,transform:'scale(.2)'}],{duration:life});
        setTimeout(()=>s.remove(), life);
      }
    }

    document.getElementById('msgBtn').onclick = ()=>{
      alert("Yahan Princee ke liye sweet birthday messages display honge ðŸ’–");
    };
